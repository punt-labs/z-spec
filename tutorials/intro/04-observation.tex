\documentclass{article}
\usepackage{fuzz}
\begin{document}

\section{Basic Types}

\begin{zed}
  [MSG, ADDR]
\end{zed}

\begin{zed}
  Flag ::= flSeen | flUnseen | flFlagged
\end{zed}

\section{Constants}

\begin{axdef}
  maxMsgs : \nat
\where
  maxMsgs = 100
\end{axdef}

\section{State}

\begin{schema}{Inbox}
  msgs : \power MSG \\
  flag : MSG \pfun Flag \\
  from : MSG \pfun ADDR
\where
  \dom flag = msgs \\
  \dom from = msgs \\
  \# msgs \leq maxMsgs
\end{schema}

\section{Observations}

\begin{schema}{CountUnseen}
  \Xi Inbox \\
  count! : \nat
\where
  count! = \# \{ m : msgs | flag~m = flUnseen \}
\end{schema}

\begin{schema}{IsEmpty}
  \Xi Inbox \\
  result! : \nat
\where
  msgs = \emptyset \implies result! = 1 \\
  msgs \neq \emptyset \implies result! = 0
\end{schema}

\begin{schema}{LookupSender}
  \Xi Inbox \\
  m? : MSG \\
  sender! : ADDR
\where
  m? \in msgs \\
  sender! = from~m?
\end{schema}
