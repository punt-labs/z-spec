\begin{document}

\section{Basic Types}

\begin{zed}
  [MSG, ADDR]
\end{zed}

\begin{zed}
  Flag ::= flSeen | flUnseen | flFlagged
\end{zed}

\section{Constants}

\begin{axdef}
  maxMsgs : \nat
\where
  maxMsgs = 100
\end{axdef}

\section{State}

\begin{schema}{Inbox}
  msgs : \power MSG \\
  flag : MSG \pfun Flag \\
  from : MSG \pfun ADDR
\where
  \dom flag = msgs \\
  \dom from = msgs \\
  \# msgs \leq maxMsgs
\end{schema}

\section{Set Operations}

\begin{schema}{CountByFlag}
  \Xi Inbox \\
  f? : Flag \\
  count! : \nat
\where
  count! = \# \{ m : msgs | flag~m = f? \}
\end{schema}

\begin{schema}{FromSender}
  \Xi Inbox \\
  sender? : ADDR \\
  result! : \power MSG
\where
  result! = \{ m : msgs | from~m = sender? \}
\end{schema}

\begin{schema}{DeleteAllSeen}
  \Delta Inbox
\where
  \LET seen == \{ m : msgs | flag~m = flSeen \} @ \\
  \quad~ msgs' = msgs \setminus seen \land \\
  \quad~ flag' = seen \ndres flag \land \\
  \quad~ from' = seen \ndres from
\end{schema}
